# Set variables used throughout the rest of the script
$PolicyName = "<fill in>"
$Domain = "<fill in>"
$Server = "<fill in>" #domain controller

## Modify variables above and then run the following command accordingly to create the GPO based on the correct AD target environment (domain/server)
New-GPO -Name $PolicyName -Comment "GPO to set cipher suite order for Windows systems (Win11/2022)" -Domain $Domain -Server $Server

## Remove the newly created GPO if there was a mistake made (uncomment and run separately)
# Remove-GPO -Name $PolicyName -Domain $Domain -Server $Server

# Set the registry value for the newer Windows system cipher suites

$params = @{
    Name      = $PolicyName
    Domain    = $Domain
    Server    = $Server
    Key       = 'HKLM\SOFTWARE\Policies\Microsoft\Cryptography\Configuration\SSL\00010002'
    ValueName = 'Functions'
    Value     = 'TLS_AES_256_GCM_SHA384','TLS_AES_128_GCM_SHA256','TLS_ECDHE_RSA_WITH_AES_256_GCM_SHA384','TLS_ECDHE_RSA_WITH_AES_128_GCM_SHA256','TLS_ECDHE_RSA_WITH_AES_256_CBC_SHA384','TLS_ECDHE_RSA_WITH_AES_128_CBC_SHA256','TLS_ECDHE_RSA_WITH_AES_256_CBC_SHA','TLS_ECDHE_RSA_WITH_AES_128_CBC_SHA','TLS_ECDHE_ECDSA_WITH_AES_256_GCM_SHA384','TLS_ECDHE_ECDSA_WITH_AES_128_GCM_SHA256','TLS_ECDHE_ECDSA_WITH_AES_256_CBC_SHA384','TLS_ECDHE_ECDSA_WITH_AES_128_CBC_SHA256','TLS_ECDHE_ECDSA_WITH_AES_256_CBC_SHA','TLS_ECDHE_ECDSA_WITH_AES_128_CBC_SHA'
    Type      = 'String'
}

Set-GPRegistryValue @params
